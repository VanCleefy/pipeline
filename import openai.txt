import openai

openai.api_key = 'YOUR_API_KEY'

response = openai.Completion.create(
  engine='text-davinci-003',  # Use the appropriate engine, e.g., 'text-davinci-003', 'text-davinci-002', etc.
  messages=[
      {"role": "system", "content": "You: Hi, let's start a new conversation."},
      {"role": "user", "content": "User: Hello, how are you?"},
      {"role": "assistant", "content": "Assistant: I'm doing well. How can I assist you today?"}
  ],
  max_tokens=100,
  temperature=0.7,
  n=1,
  stop=None
)

print(response.choices[0].message.content)
